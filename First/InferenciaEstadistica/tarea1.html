      return(mean(x[1:i]))

Posteriormente simulamos los datos, en particular cada \(\bar{X_n}\) con \(i\in\{1,5,25,100\}\) se simuló 300 veces.

set.seed(0)
sampling <- function(i)
{
  x1 <- runif(i, 0, 1)
  if(i==1){
    z <- c(x1, x1)
    names(z) <- c('media', 'var')
    return(z)
  }
  else {
    z <- c(mean(x1), var(x1))
    names(z) <- c('media', 'var')
  return( z)
  }
}

m <- 300
x_1 <- mapply( sampling, rep(1, m ))
x_1 <- data.frame(t(x_1))
